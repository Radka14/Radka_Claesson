SELECT
    branch,
    year,
    avg_salary,
    ROUND(
        (avg_salary - LAG(avg_salary) OVER (PARTITION BY branch ORDER BY year)) 
        / LAG(avg_salary) OVER (PARTITION BY branch ORDER BY year) * 100, 2
    ) AS salary_change_percent,
    CASE 
        WHEN (avg_salary - LAG(avg_salary) OVER (PARTITION BY branch ORDER BY year)) > 0 THEN 'Růst'
        WHEN (avg_salary - LAG(avg_salary) OVER (PARTITION BY branch ORDER BY year)) < 0 THEN 'Pokles'
        ELSE 'Beze změny'
    END AS trend
FROM t_radka_claesson_project_SQL_primary_final
GROUP BY branch, year, avg_salary
ORDER BY branch, year;
